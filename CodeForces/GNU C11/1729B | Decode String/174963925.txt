#include <stdio.h>
#include <string.h>
 
int q, n;
char s[60];
 
char *strrev(char *str)
{
      char *p1, *p2;
 
      if (! str || ! *str)
            return str;
      for (p1 = str, p2 = str + strlen(str) - 1; p2 > p1; ++p1, --p2)
      {
            *p1 ^= *p2;
            *p2 ^= *p1;
            *p1 ^= *p2;
      }
      return str;
}
 
int main(){
    scanf("%d", &q);
    while(q --){
        scanf("%d", &n);
        scanf("%s", s);
        char ans[60] = "";
        int sz = 0;
        for(int i = n - 1 ; i >= 0 ; i --){
            if(s[i] == '0')
                ans[sz++] = (char)((s[i - 1] - '1') + 10 * (s[i - 2] - '0') + 'a'), i -= 2;
 
            else
                ans[sz++] = (char)(s[i] - '1' + 'a');
        }
        printf("%s\n", strrev(ans));
    }
    return 0;
}
